        2) Display all employees:

           db.Employee.find()

        3) Find the employees whose salary is >10000:

           db.Employee.find({salary:{$gt:10000}})

        4) Display the employee name whose having salary  either 15000 or 25000

           db.Employee.find({$or:[{salary:15000},{salary:25000}]})

        5) Update designation of employee 11 to ‘Asst. Prof.’

           db.Employee.updateOne({empid:11},{$set:{designation:"Asst.Prof"}})

        6) Delete the employees having designation “supervisor”.

           db.Employee.deleteMany({designation:"supervisor"});

        7) Display top 2 employees having highest salary.

           db.Employee.find().sort({salary:-1}).limit(2)

        8) Increment all employee salary by 1000.

           db.Employee.updateMany({},{$inc:{salary:1000}})

        9) Increment all employee salary by 6%.

           db.Employee.updateMany({},{$mul:{salary:1.06}})
