1)Write PLSQL function to find reverse of given String.


        CREATE OR REPLACE FUNCTION rev(word VARCHAR) 
        RETURN VARCHAR IS 
        len NUMBER; 
        str1 VARCHAR(20); 
        BEGIN 
        len := Length(word); 
          FOR i IN REVERSE 1.. len LOOP            
            str1 := str1|| Substr(word, i, 1); 
          END LOOP; 
          return str1; 
        END;

        dbms_output.put_line(rev('Shreyash')); 
        END;

        CREATE OR REPLACE PROCEDURE findAvg(num1 NUMBER,num2 NUMBER, num3 NUMBER) 
        AS 
        sumNum Number; 
        ans NUMBER(10,4); 
        BEGIN 
            sumNum:=num1+num2+num3; 
            ans:=sumNum/3; 
            dbms_output.put_line(ans); 
        END;
